	var notificationData = {
		cancelIndex: 0,
		buttonNames: [formCancel,formConfirm],
		fields: formFields,
		bookName: bookName,
		formAppHighUsage: formAppHighUsage,
		formName: formName,

			//START IF - Cancel Clicked close ELSE

		    };

		},

	};
	// END Loop - through fields
